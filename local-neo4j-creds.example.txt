# Sample Neo4j credentials for the local Docker Compose stack.
# Copy this file to local-neo4j-creds.txt and upload it via the app sidebar,
# or paste the values into the manual form fields.
#
# Docker Compose: NEO4J_URI uses the `neo4j` service name as the host so the
# backend container can reach it on the internal network.
# AuraDB: replace NEO4J_URI with the `neo4j+s://...` URI from your Aura
# console and update the username/password/database to match.
NEO4J_URI=neo4j://neo4j:7687
NEO4J_USERNAME=neo4j
NEO4J_PASSWORD=devpassword
NEO4J_DATABASE=neo4j

# Optional, if using AuraDB rather than local Neo4j:
AURA_INSTANCEID=instanceid
AURA_INSTANCENAME=instancename
